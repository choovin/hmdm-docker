# Feature 9: 设备联系人管理 (Device Contacts Management)

## 1. 功能概述

设备联系人管理功能允许管理员集中管理设备的联系人信息，支持批量添加、更新、删除联系人，并可同步到指定设备。

### 核心功能
- 联系人CRUD管理
- 批量导入联系人
- 设备联系人同步
- 联系人分组
- 联系人搜索
- 同步状态跟踪

## 3. 数据库设计

### 3.1 设备联系人表 (device_contacts)

```sql
CREATE TABLE device_contacts (
    id SERIAL PRIMARY KEY,
    device_id INTEGER REFERENCES devices(id) ON DELETE CASCADE,
    customer_id INTEGER NOT NULL,

    -- 联系人信息
    display_name VARCHAR(255) NOT NULL,
    first_name VARCHAR(100),
    last_name VARCHAR(100),

    -- 联系方式
    phone_numbers TEXT, -- JSON数组 [{"type": "MOBILE", "number": "+86138..."}]
    emails TEXT, -- JSON数组 [{"type": "WORK", "email": "..."}]

    -- 组织信息
    company VARCHAR(255),
    job_title VARCHAR(100),
    department VARCHAR(100),

    -- 其他信息
    notes TEXT,
    groups TEXT, -- JSON数组 ["同事", "客户"]

    -- 头像
    photo_url VARCHAR(500),

    -- 同步状态
    sync_status VARCHAR(20) DEFAULT 'PENDING', -- PENDING, SYNCED, ERROR
    last_synced_at TIMESTAMP,
    sync_error TEXT,

    -- 来源
    source VARCHAR(20) DEFAULT 'ADMIN', -- ADMIN, IMPORT, DEVICE

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    INDEX idx_device (device_id),
    INDEX idx_sync_status (sync_status),
    INDEX idx_display_name (display_name)
);
```

## 6. 关键代码示例

### 6.1 联系人同步服务

```java
@Singleton
public class DeviceContactService {
    private final DeviceContactMapper contactMapper;
    private final PushService pushService;

    @Transactional
    public DeviceContact createContact(DeviceContact contact) {
        contact.setCustomerId(getCurrentCustomerId());
        contact.setSyncStatus("PENDING");
        contact.setSource("ADMIN");

        contactMapper.insert(contact);

        // 如果需要立即同步
        if (contact.getDeviceId() != null) {
            syncContactToDevice(contact.getDeviceId(), contact);
        }

        return contact;
    }

    public void syncContactsToDevice(Integer deviceId) {
        List<DeviceContact> pendingContacts = contactMapper
            .getPendingContacts(deviceId, getCurrentCustomerId());

        if (pendingContacts.isEmpty()) {
            return;
        }

        // 构建同步数据
        ContactsSyncData syncData = new ContactsSyncData();
        syncData.setOperation("SYNC");
        syncData.setContacts(convertToSyncFormat(pendingContacts));

        // 发送推送通知
        boolean sent = pushService.sendMessage(
            deviceId,
            PushMessage.TYPE_CONTACTS_SYNC,
            JsonMapper.toJson(syncData)
        );

        if (sent) {
            // 更新同步状态
            for (DeviceContact contact : pendingContacts) {
                contactMapper.updateSyncStatus(
                    contact.getId(), "SYNCING", null
                );
            }
        }
    }

    private List<ContactSyncItem> convertToSyncFormat(List<DeviceContact> contacts) {
        List<ContactSyncItem> items = new ArrayList<>();

        for (DeviceContact contact : contacts) {
            ContactSyncItem item = new ContactSyncItem();
            item.setId(contact.getId());
            item.setDisplayName(contact.getDisplayName());
            item.setPhoneNumbers(parsePhoneNumbers(contact.getPhoneNumbers()));
            item.setEmails(parseEmails(contact.getEmails()));
            item.setCompany(contact.getCompany());
            item.setJobTitle(contact.getJobTitle());
            item.setNotes(contact.getNotes());
            items.add(item);
        }

        return items;
    }
}
```

### 6.2 Android Agent端联系人同步

```java
// ContactsSyncReceiver.java
public class ContactsSyncReceiver extends BroadcastReceiver {
    @Override
    public void onReceive(Context context, Intent intent) {
        if (PushMessage.TYPE_CONTACTS_SYNC.equals(
            intent.getStringExtra("messageType"))) {

            String payload = intent.getStringExtra("payload");
            handleContactsSync(context, payload);
        }
    }

    private void handleContactsSync(Context context, String payload) {
        try {
            JSONObject data = new JSONObject(payload);
            String operation = data.getString("operation");
            JSONArray contacts = data.getJSONArray("contacts");

            if ("SYNC".equals(operation)) {
                for (int i = 0; i < contacts.length(); i++) {
                    JSONObject contactJson = contacts.getJSONObject(i);
                    syncContactToPhone(context, contactJson);
                }

                // 上报同步结果
                reportSyncResult(context, data.getInt("syncId"), true, null);
            } else if ("DELETE".equals(operation)) {
                // 删除联系人
                deleteContacts(context, contacts);
            }

        } catch (Exception e) {
            Log.e(TAG, "Failed to sync contacts", e);
            reportSyncResult(context, -1, false, e.getMessage());
        }
    }

    private void syncContactToPhone(Context context, JSONObject contactJson) {
        ArrayList<ContentProviderOperation> ops = new ArrayList<>();

        int rawContactInsertIndex = ops.size();

        // 添加RawContact
        ops.add(ContentProviderOperation.newInsert(RawContacts.CONTENT_URI)
            .withValue(RawContacts.ACCOUNT_TYPE, null)
            .withValue(RawContacts.ACCOUNT_NAME, null)
            .build());

        // 添加姓名
        ops.add(ContentProviderOperation.newInsert(Data.CONTENT_URI)
            .withValueBackReference(Data.RAW_CONTACT_ID, rawContactInsertIndex)
            .withValue(Data.MIMETYPE, StructuredName.CONTENT_ITEM_TYPE)
            .withValue(StructuredName.DISPLAY_NAME, contactJson.optString("displayName"))
            .withValue(StructuredName.GIVEN_NAME, contactJson.optString("firstName"))
            .withValue(StructuredName.FAMILY_NAME, contactJson.optString("lastName"))
            .build());

        // 添加电话号码
        JSONArray phones = contactJson.optJSONArray("phoneNumbers");
        if (phones != null) {
            for (int i = 0; i < phones.length(); i++) {
                JSONObject phone = phones.getJSONObject(i);
                ops.add(ContentProviderOperation.newInsert(Data.CONTENT_URI)
                    .withValueBackReference(Data.RAW_CONTACT_ID, rawContactInsertIndex)
                    .withValue(Data.MIMETYPE, Phone.CONTENT_ITEM_TYPE)
                    .withValue(Phone.NUMBER, phone.getString("number"))
                    .withValue(Phone.TYPE, getPhoneType(phone.optString("type")))
                    .build());
            }
        }

        // 添加邮箱
        JSONArray emails = contactJson.optJSONArray("emails");
        if (emails != null) {
            for (int i = 0; i < emails.length(); i++) {
                JSONObject email = emails.getJSONObject(i);
                ops.add(ContentProviderOperation.newInsert(Data.CONTENT_URI)
                    .withValueBackReference(Data.RAW_CONTACT_ID, rawContactInsertIndex)
                    .withValue(Data.MIMETYPE, Email.CONTENT_ITEM_TYPE)
                    .withValue(Email.ADDRESS, email.getString("email"))
                    .withValue(Email.TYPE, getEmailType(email.optString("type")))
                    .build());
            }
        }

        // 添加组织信息
        if (contactJson.has("company") || contactJson.has("jobTitle")) {
            ops.add(ContentProviderOperation.newInsert(Data.CONTENT_URI)
                .withValueBackReference(Data.RAW_CONTACT_ID, rawContactInsertIndex)
                .withValue(Data.MIMETYPE, Organization.CONTENT_ITEM_TYPE)
                .withValue(Organization.COMPANY, contactJson.optString("company"))
                .withValue(Organization.TITLE, contactJson.optString("jobTitle"))
                .build());
        }

        // 执行批量操作
        try {
            context.getContentResolver().applyBatch(ContactsContract.AUTHORITY, ops);
        } catch (Exception e) {
            Log.e(TAG, "Failed to insert contact", e);
            throw new RuntimeException(e);
        }
    }

    private int getPhoneType(String type) {
        switch (type) {
            case "MOBILE": return Phone.TYPE_MOBILE;
            case "WORK": return Phone.TYPE_WORK;
            case "HOME": return Phone.TYPE_HOME;
            case "FAX": return Phone.TYPE_FAX_WORK;
            default: return Phone.TYPE_OTHER;
        }
    }

    private void reportSyncResult(Context context, int syncId,
                                  boolean success, String error) {
        JSONObject result = new JSONObject();
        try {
            result.put("syncId", syncId);
            result.put("success", success);
            if (error != null) {
                result.put("error", error);
            }

            ServerConnection.sendContactsSyncResult(result);
        } catch (JSONException e) {
            Log.e(TAG, "Failed to report sync result", e);
        }
    }
}
```

## 7. 安全考虑

1. **权限控制**: 需要WRITE_CONTACTS权限
2. **数据隐私**: 联系人数据加密存储
3. **同步冲突**: 处理设备本地联系人与服务器联系人的冲突
4. **数据备份**: 联系人在同步前备份
