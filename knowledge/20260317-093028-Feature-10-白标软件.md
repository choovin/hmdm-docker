# Feature 10: 白标软件 (White-Label Software)

## 1. 功能概述

白标软件功能允许企业定制自己的MDM应用，包括应用图标、名称、颜色主题、服务器URL等，打造专属的企业品牌MDM解决方案。

### 核心功能
- 应用名称和图标定制
- 颜色主题定制
- 启动画面定制
- 服务器URL预配置
- 签名证书管理
- APK自动构建
- 构建历史管理
- 邮件模板定制

```sql
    -- 品牌信息

    -- 功能开关

    allow_server_change BOOLEAN DEFAULT FALSE,
    show_help_link BOOLEAN DEFAULT TRUE,

    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
```

### 3.2 白标构建记录表 (white_label_builds)

```sql
CREATE TABLE white_label_builds (
    id SERIAL PRIMARY KEY,
    customer_id INTEGER NOT NULL,
    settings_id INTEGER REFERENCES white_label_settings(id),

    -- 构建信息
    build_number INTEGER NOT NULL,
    version_name VARCHAR(50),
    version_code INTEGER,

    -- 构建状态: PENDING, BUILDING, SUCCESS, FAILED
    status VARCHAR(20) DEFAULT 'PENDING',

    -- 文件信息
    apk_url VARCHAR(500),
    apk_size BIGINT,
    apk_checksum VARCHAR(64),

    -- 签名信息
    is_signed BOOLEAN DEFAULT FALSE,
    keystore_alias VARCHAR(100),

    -- 时间戳
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    started_at TIMESTAMP,
    completed_at TIMESTAMP,

    -- 构建日志
    build_log TEXT,
    error_message TEXT,

    -- 构建者
    created_by INTEGER REFERENCES users(id),

    INDEX idx_customer_status (customer_id, status),
    INDEX idx_created_at (created_at)
);
```

### 6.1 APK构建服务

```java
@Singleton
public class WhiteLabelBuildService {
    private final WhiteLabelSettingsMapper settingsMapper;
    private final WhiteLabelBuildMapper buildMapper;
    private final FileStorageService fileStorage;

    @Transactional
    public WhiteLabelBuild createBuild(WhiteLabelBuild buildRequest) {
        WhiteLabelSettings settings = settingsMapper.getSettings();

        // 生成版本号
        int nextBuildNumber = buildMapper.getNextBuildNumber();

        WhiteLabelBuild build = new WhiteLabelBuild();
        build.setCustomerId(getCurrentCustomerId());
        build.setSettingsId(settings.getId());
        build.setBuildNumber(nextBuildNumber);
        build.setVersionName(buildRequest.getVersionName());
        build.setVersionCode(buildRequest.getVersionCode());
        build.setStatus("PENDING");
        build.setCreatedBy(getCurrentUserId());
        build.setCreatedAt(System.currentTimeMillis());

        buildMapper.insert(build);

        // 异步触发构建
        triggerBuildAsync(build.getId());

        return build;
    }

    @Async
    public void triggerBuildAsync(Integer buildId) {
        WhiteLabelBuild build = buildMapper.getById(buildId);
        WhiteLabelSettings settings = settingsMapper.getById(build.getSettingsId());

        try {
            buildMapper.updateStatus(buildId, "BUILDING", null, null);

            // 准备构建资源
            BuildResources resources = prepareBuildResources(settings);

            // 执行构建
            BuildResult result = executeBuild(build, resources);

            if (result.isSuccess()) {
                // 上传APK
                String apkUrl = fileStorage.store(result.getApkFile());

                buildMapper.updateStatus(buildId, "SUCCESS", apkUrl, null);
                buildMapper.updateApkInfo(buildId,
                    result.getApkSize(),
                    result.getApkChecksum());

                // 签名APK（如果需要）
                if (build.getIsSigned()) {
                    signApk(buildId, build.getKeystoreAlias());
                }
            } else {
                buildMapper.updateStatus(buildId, "FAILED", null, result.getErrorMessage());
            }

        } catch (Exception e) {
            buildMapper.updateStatus(buildId, "FAILED", null, e.getMessage());
        }
    }

    private BuildResult executeBuild(WhiteLabelBuild build, BuildResources resources) {
        try {
            // 创建工作目录
            Path workDir = Files.createTempDirectory("whitelabel-build-" + build.getId());

            // 1. 克隆基础项目代码
            cloneBaseProject(workDir);

            // 2. 替换资源文件
            replaceResources(workDir, resources);

            // 3. 修改配置文件
            updateConfigFiles(workDir, resources.getSettings());

            // 4. 执行构建
            ProcessBuilder pb = new ProcessBuilder(
                "./gradlew",
                "assembleRelease",
                "-PVERSION_NAME=" + build.getVersionName(),
                "-PVERSION_CODE=" + build.getVersionCode()
            );
            pb.directory(workDir.toFile());
            pb.inheritIO();

            Process process = pb.start();
            int exitCode = process.waitFor();

            if (exitCode == 0) {
                // 找到生成的APK
                Path apkPath = workDir.resolve("app/build/outputs/apk/release/app-release-unsigned.apk");

                return BuildResult.success(apkPath.toFile());
            } else {
                return BuildResult.fail("Build failed with exit code: " + exitCode);
            }

        } catch (Exception e) {
            return BuildResult.fail(e.getMessage());
        }
    }

    private void replaceResources(Path workDir, BuildResources resources) throws IOException {
        // 替换应用图标
        if (resources.getAppIcon() != null) {
            Path iconDir = workDir.resolve("app/src/main/res");
            generateIcons(resources.getAppIcon(), iconDir);
        }

        // 替换启动画面
        if (resources.getSplashLogo() != null) {
            Path splashDir = workDir.resolve("app/src/main/res/drawable");
            Files.copy(resources.getSplashLogo().toPath(),
                splashDir.resolve("splash_logo.png"),
                StandardCopyOption.REPLACE_EXISTING);
        }

        // 替换公司Logo
        if (resources.getCompanyLogo() != null) {
            Path drawableDir = workDir.resolve("app/src/main/res/drawable");
            Files.copy(resources.getCompanyLogo().toPath(),
                drawableDir.resolve("company_logo.png"),
                StandardCopyOption.REPLACE_EXISTING);
        }
    }

    private void updateConfigFiles(Path workDir, WhiteLabelSettings settings) throws IOException {
        // 更新strings.xml
        Path stringsXml = workDir.resolve("app/src/main/res/values/strings.xml");
        String content = new String(Files.readAllBytes(stringsXml), StandardCharsets.UTF_8);

        content = content.replace("{{APP_NAME}}", settings.getAppName());
        content = content.replace("{{COMPANY_NAME}}", settings.getCompanyName());
        content = content.replace("{{SERVER_URL}}", settings.getServerUrl());
        content = content.replace("{{COPYRIGHT}}", settings.getCopyrightText());

        Files.write(stringsXml, content.getBytes(StandardCharsets.UTF_8));

        // 更新colors.xml
        Path colorsXml = workDir.resolve("app/src/main/res/values/colors.xml");
        String colorsContent = "<?xml version=\"1.0\" encoding=\"utf-8\"?>\n" +
            "<resources>\n" +
            "    <color name=\"colorPrimary\">" + settings.getPrimaryColor() + "</color>\n" +
            "    <color name=\"colorAccent\">" + settings.getAccentColor() + "</color>\n" +
            "    <color name=\"statusBarColor\">" + settings.getStatusBarColor() + "</color>\n" +
            "    <color name=\"splashBackground\">" + settings.getSplashBackgroundColor() + "</color>\n" +
            "</resources>";

        Files.write(colorsXml, colorsContent.getBytes(StandardCharsets.UTF_8));

        // 更新AndroidManifest.xml中的服务器URL
        Path manifestXml = workDir.resolve("app/src/main/AndroidManifest.xml");
        String manifestContent = new String(Files.readAllBytes(manifestXml), StandardCharsets.UTF_8);

        manifestContent = manifestContent.replace(
            "{{DEFAULT_SERVER_URL}}",
            settings.getServerUrl()
        );

        Files.write(manifestXml, manifestContent.getBytes(StandardCharsets.UTF_8));
    }

    private void signApk(Integer buildId, String keystoreAlias) {
        // 获取APK和密钥库
        WhiteLabelBuild build = buildMapper.getById(buildId);
        Keystore keystore = keystoreMapper.getByAlias(keystoreAlias);

        try {
            // 使用apksigner或jarsigner签名
            Path unsignedApk = fileStorage.getPath(build.getApkUrl());
            Path signedApk = Files.createTempFile("signed-", ".apk");

            ProcessBuilder pb = new ProcessBuilder(
                "apksigner",
                "sign",
                "--ks", keystore.getFilePath(),
                "--ks-key-alias", keystoreAlias,
                "--ks-pass", "pass:" + keystore.getPassword(),
                "--key-pass", "pass:" + keystore.getKeyPassword(),
                "--out", signedApk.toString(),
                unsignedApk.toString()
            );

            Process process = pb.start();
            process.waitFor();

            // 更新为签名后的APK
            String signedApkUrl = fileStorage.store(signedApk.toFile());
            buildMapper.updateApkUrl(buildId, signedApkUrl);
            buildMapper.updateSignedStatus(buildId, true);

        } catch (Exception e) {
            Log.e(TAG, "Failed to sign APK", e);
        }
    }
}
```

```java
@Singleton
public class EmailTemplateService {
    private final EmailTemplateMapper templateMapper;

    public void sendEmail(String templateType, String toEmail,
                          Map<String, String> variables) {
        EmailTemplate template = templateMapper.getTemplate(
            getCurrentCustomerId(), templateType);

        if (template == null || !template.getIsEnabled()) {
            // 使用默认模板
            template = getDefaultTemplate(templateType);
        }

        // 替换变量
        String subject = replaceVariables(template.getSubject(), variables);
        String htmlBody = replaceVariables(template.getBodyHtml(), variables);
        String textBody = replaceVariables(template.getBodyText(), variables);

        // 发送邮件
        EmailMessage message = new EmailMessage();
        message.setFrom(template.getFromEmail(), template.getFromName());
        message.setTo(toEmail);
        message.setSubject(subject);
        message.setHtmlBody(htmlBody);
        message.setTextBody(textBody);

        emailService.send(message);
    }

    private String replaceVariables(String template, Map<String, String> variables) {
        String result = template;
        for (Map.Entry<String, String> entry : variables.entrySet()) {
            result = result.replace("{{" + entry.getKey() + "}}", entry.getValue());
        }
        return result;
    }
}
```
